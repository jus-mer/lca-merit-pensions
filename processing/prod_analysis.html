4.2 Latent class analysis with dichotomized items

Show the code 
# ---- build table (no list-cols) -----------------------------------------

N_used <- nrow(db_lca)

m_patterns <- n_patterns_from_data(db_lca)  # scalar


# Robust extractors (avoid list-columns)
get_K <- function(fit) as.integer(ncol(fit$posterior))  # classes = cols of posterior
get_ll  <- function(fit) as.numeric(fit$llik)
get_aic <- function(fit) as.numeric(fit$aic)
get_bic <- function(fit) as.numeric(fit$bic)
get_npar <- function(fit) {
  if (!is.null(fit$npar)) return(as.numeric(fit$npar))
  # fallback: compute from probs
  K <- get_K(fit)
  Rj <- vapply(fit$probs, function(m) ncol(m), integer(1))
  as.numeric(sum(K * (Rj - 1)) + (K - 1))
}

# Table of fit indices
fit_tbl <- tibble(
  K     = map_int(fits, get_K),
  N     = nrow(db_lca),
  logLik= map_dbl(fits, get_ll),
  npar  = map_dbl(fits, get_npar),
  AIC   = map_dbl(fits, get_aic),
  BIC   = map_dbl(fits, get_bic)
) %>%
  arrange(K) %>%
  mutate(
    # Differences relative to preceding (K-1) model
    dAIC = AIC - lag(AIC),
    dBIC = BIC - lag(BIC),
    
    # Likelihood-ratio test (K vs K-1)
    # Test statistic: 2*(LL_K - LL_{K-1}) ~ Chi-square with df = npar_K - npar_{K-1}
    # Note: in LCA, models are not strictly nested in the regular sense;
    # this LRT is widely used as a heuristic, but interpret with caution.
    LRT = 2 * (logLik - lag(logLik)),
    df_LRT = npar - lag(npar),
    p_LRT = ifelse(!is.na(LRT) & df_LRT > 0, pchisq(LRT, df = df_LRT, lower.tail = FALSE), NA_real_)
  )


fit_tbl_report <- fit_tbl %>%
  transmute(
    K, N,
    logLik = round(logLik, 1),
    npar = npar,
    AIC = round(AIC, 1),
    dAIC = round(dAIC, 1),
    BIC = round(BIC, 1),
    dBIC = round(dBIC, 1),
    LRT = round(LRT, 1),
    df_LRT = df_LRT,
    p_LRT = signif(p_LRT, 3)
  )

The Table 3 shows the conditional probabilities of each item associated with a latent class and their respective sizes. Class one has a size of 36.3% and shows low probabilities of perceiving meritocratic norms in society, while it highly perceives non-meritocratic functions. In addition, there is a high preference for both meritocratic and non-meritocratic elements, with talent being the only factor that deviates slightly from this pattern. In short, this class does not perceive a meritocratic functioning in society, and at the same time adheres to both meritocratic and non-meritocratic factors. This class is referred to as “Anything goes preferences.”

In the second class, there is a high perception of meritocratic elements and conditions of origin at the societal level. On the other hand, there is a strong preference for effort and talent, which differs from a lower adherence to non-meritocratic rules. In this case, there is a panoramic view of how society works, where effort and talent coexist with non-meritocratic functions; however, people prefer to be rewarded primarily for their merit. This group includes “Meritocratic believers with a broad perspective.”

The fourth class shows almost no perception of meritocratic norms in society, while perceiving a social order mainly governed by conditions of origin. In relation to this, preferences show a high adherence to the idea that society should function on the basis of meritocratic elements (although effort rather than talent), but having rich parents and good contacts should not influence the distribution of rewards in society. In this context, there is a clear narrative where, perceiving that society operates on the basis of non-meritocratic elements, there is a preference for meritocracy to come into play. This class is designated as “Radical meritocrats.”

# ---- 2) Class size + separation diagnostics

sep_diag <- function(fit) {
  P <- fit$posterior
  K <- ncol(P)
  class_hat <- max.col(P)
  max_p <- apply(P, 1, max)
  
  avg_diag <- vapply(1:K, function(k) {
    idx <- which(class_hat == k)
    if (length(idx) == 0) NA_real_ else mean(P[idx, k])
  }, numeric(1))
  
  tibble(
    K = K,
    min_class_prior = min(fit$P),
    class_prior = paste0(round(100 * fit$P, 1), collapse = " / "),
    mean_max_p = mean(max_p),
    p10_max_p  = unname(quantile(max_p, 0.10)),
    avg_diag_min  = suppressWarnings(min(avg_diag, na.rm = TRUE)),
    avg_diag_mean = mean(avg_diag, na.rm = TRUE)
  )
}

# determine endorsement category from your factors
endorsement_level <- tail(levels(db_lca$a1), 1)

get_p_endorse <- function(fit, endorsement_level) {
  items <- names(fit$probs)
  
  probs_long <- bind_rows(lapply(items, function(item) {
    mat <- fit$probs[[item]]              # K x 2 (dichotomous)
    df <- as.data.frame(mat)
    df$class <- 1:nrow(mat)
    df$item <- item
    df
  })) %>%
    pivot_longer(cols = -c(class, item), names_to = "category", values_to = "prob") %>%
    mutate(class = as.integer(class), prob = as.numeric(prob)) %>%
    filter(category == endorsement_level) %>%
    transmute(item, class, p = prob)
  
  probs_long
}
